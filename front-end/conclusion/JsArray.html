    # JS数组总结

    # 1、定义数组

    <!-- 方式一 -->
const arr1 = new Array("banana", 1, true);

<!-- 方式二 -->
consst arr2 = ["banana", 1, true]

    # 2、方法

    # 长度

    # length()

    • 获取数组长度
    const arr = [1,2,3,4,5]
console.log(arr.length) // 5

const arr = Array(6) // 使用此方法创建，数组中会有6个元素，值为undefined
const arr = Array.of(6) // 此方法创建，数组只会有一个元素，值为6

    # 数组与字符串转换

    # String()/toString()

    • 将数组转化为字符串
    const arr = [1,2,3,4,5]
const a1 = String(arr)
const a2 = arr.toString()
console.log(a1,a2) // string,string

    # join()

    • 将数组以某个字符进行连接
    const arr = [1,2,3,4,5].join('-')
console.log(arr) // 1-2-3-4-5

    # Array.from()

    • 将字符串转化为数组
    const str = 'hello'
console.log(Array.from(str)) // ['h','e','e','l','l','o']

    # 追加/合并

    # [...arr1,...arr2]

    • 向数组中追加元素
    • 将一个类数组转化为真正的数组
    const arr1 = [1,2,3]
const arr2 = [12,23]
arr1 = [...arr1,...arr2]
console.log(arr1) // [1,2,3,12,23]

    # contact()

    • 合并两个数组
    • 返回值：合并后的新数组
    • 不会改变原数组
    • 可以合并多个数组
    const arr1 = [1,2,3]
const arr2 = [12,23]
const newArr = arr1.contact(arr2)
console.log(newArr) // [1,2,3,12,23]

    # 添加

    # push()

    • 向数组最后添加元素
    • 返回值：数组长度
    • 会改变原始数组
    const arr = [1,2]
let len = arr.push(3,4)
console.log(arr,len) // [1,2,3,4] 4

    # unshift()

    • 向数组前面添加元素
    • 返回值：数组长度
    • 会改变原始数组
    const arr = [1,2]
let len = arr.unshift(3,4)
console.log(arr,len) // [3, 4, 1, 2] 4

    # 删除

    # pop()

    • 删除数组中最后一个元素
    • 返回值：删除的那一个元素
    • 会改变原始数组
    const arr = [1,2,3,4,5]
let val = arr.pop()
console.log(arr,val) // [1,2,3,4] 5

    # shift()

    • 删除数组中前面一个元素
    • 返回值：删除的那一个元素
    • 会改变原始数组
    const arr = [1,2,3,4,5]
let val = arr.shift()
console.log(arr,val) // [2,3,4,5] 1

    # 替换

    # fill(value, start, end)

    • 替换数组中指定位置的元素（左闭右开）
    • 返回值：修改后的数组
    • 会改变原始数组
    const arr = [1,2]
let val = arr.fille(12,1,2)
console.log(arr,val) // [1,12] [1,12]

    # 截取

    # slice(start,end)

    • 截取数组指定位置元素（左闭右开）
    • 返回值：截取的元素
    • 不会改变原数组
    const arr = [1, 2, 3, 4, 5];
let val = arr.slice(1,3);
console.log(arr, val); // [1, 2, 3, 4, 5] [2, 3]

    # splice(start,len)

    • 种类
      • 从数组指定位置截取固定长度的元素splice(1,3)
      • 删除数组中的某一个元素splice(1,1)
      • 替换数组中的元素splice(1,1,val)
      • 向数组中追加元素splice(1,0,val)
        • 向数组开头添加元素splice(0,0,val)
        • 向数组末尾添加元素splice(arr.length,0,val)
    • 返回值：截取的元素
    • 会改变原始数组
    const arr = [1, 2, 3, 4, 5];
let val = arr.splice(1,3);
console.log(arr, val); // [1, 5] [2, 3, 4]

    # 复制

    # copyWithin(target,start,end)

    • 将数组中的某些元素复制到指定位置
    • 不会改变原数组
    const arr = [1, 2, 3, 4, 5];
const newArray = arr.copyWithin(2,1,2)
console.log(newArray) // [1, 2, 2, 4, 5]

    # 查找

    # indexOf(num,start)

    • 查找数组中元素的位置（从左侧开始）
    • 返回值：元素在数组中的索引
    • 当查找的值不存在时，返回*-1*
    • start为负数时，表示从右侧开始查找
    const arr = [1,2,3,4,5]
console.log(arr.indexOf(3),arr.indexOf(10)) // 2 -1

    # lastIndexOf(num,start)

    • 查找数组中元素的位置（从右侧开始）
    • start为负数时，表示从右侧开始查找

    # includes(num)

    • 查找数组中是否存在某个值
    • 返回值true或false
    const arr = [1,2,3,4,5]
console.log(arr.includes(3),arr.includes(10)) // true false

    # find()/findIndex()

    • 查找数组中是否有某个元素
    • 有：返回该值；没有：返回undefined
    • 不改变原数组
    • findIndex()返回元素索引
    const arr = [1, 2, 3, 4, 5];
const val = arr.find(item=>{
  return item === 3
})
console.log(val);  // 3

    # 排序

    # sort()

    • 对数组元素进行排序
    • 比较两个值，后者减去前者是从大到小，前者减去后者是从小到大
    • 会改变原数组
    const arr = [12, 32, 14, 53, 12];
const val = arr.sort((a, b) => {
  return b-a;
});
console.log(val); // [53, 32, 14, 12, 12]

    # 循环

    # for ... in

    • 输出索引值
    const phone = [
  { id: 101, brand: "苹果", price: 4599 },
  { id: 102, brand: "华为", price: 2399 },
  { id: 103, brand: "小米", price: 1799 },
  { id: 104, brand: "魅族", price: 2799 },

for(let val in phone){
  console.log(val);
}

    # for ... of

    • 输出每个值
    const phone = [
  { id: 101, brand: "苹果", price: 4599 },
  { id: 102, brand: "华为", price: 2399 },
  { id: 103, brand: "小米", price: 1799 },
  { id: 104, brand: "魅族", price: 2799 },

for(let val of phone){
  console.log(val);
}

    # forEach()

    • 循环遍历数组中的每条数据
    const phone = [
  { id: 101, brand: "苹果", price: 4599 },
  { id: 102, brand: "华为", price: 2399 },
  { id: 103, brand: "小米", price: 1799 },
  { id: 104, brand: "魅族", price: 2799 },
];
phone.forEach((item, index, arr) => {
  console.log(item, index, arr);
});

    # every()

    • 判断数组中的数据是否都符合条件
    const phone = [
  { id: 101, brand: "苹果", price: 4599 },
  { id: 102, brand: "华为", price: 2399 },
  { id: 103, brand: "小米", price: 1799 },
  { id: 104, brand: "魅族", price: 2799 },
];
let state = phone.every((item) => {
  return item.price <= 2500;
});
console.log(state); // fslse

    # some()

    • 判断数组中的某一个元素是否都符合条件
    • 当有一个为真，就停止检测
    const phone = [
  { id: 101, brand: "苹果", price: 4599 },
  { id: 102, brand: "华为", price: 2399 },
  { id: 103, brand: "小米", price: 1799 },
  { id: 104, brand: "魅族", price: 2799 },
];
let state = phone.some((item) => {
  return item.price <= 2500;
});
console.log(state);  // true

    # 映射

  { id: 102, brand: "华为", price: 2399 },
  { id: 103, brand: "小米", price: 1799 },

phone.map((item) => {
   item.price = "¥" + item.price;
});
console.table(phone);

    # reduce()

    • 接收一个函数作为累加器，数组中的每个值（从左到右）开始缩减，最终计算为一个值
    const phone = [
  { id: 101, brand: "苹果", price: 4599 },
  { id: 102, brand: "华为", price: 2399 },
  { id: 103, brand: "小米", price: 1799 },
  { id: 104, brand: "魅族", price: 2799 },
  { id: 105, brand: "苹果", price: 8799 },
  { id: 106, brand: "魅族", price: 2799 },
  { id: 107, brand: "华为", price: 9799 },
  { id: 108, brand: "魅族", price: 2799 },
];

let num = phone.reduce(
  (origin, item) => {
    if (item.brand === "苹果") {
      origin.arr.push(item);
      origin.total += 1;
      origin.brand = item.brand;
    }
    return origin;
  },
  { arr: [], total: 0 }
);
console.log(num);
