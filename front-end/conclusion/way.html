    # 组件的通信方式

    # 父子组件通信

    # 父传子/子传父

    # 方式一：props传参

    父组件

    <template>
	<!-- getDemoNum是子传父，total是父传子 -->
  <demo :getDemoNum="getDemoNum" :total="20" />
</template>
<script>
import { reactive, toRefs } from "vue";
import Demo from "./components/Demo.vue";
export default {
  name: "App",
  components: {
    Demo,
  },
  setup() {
    const state = reactive({});
    const getDemoNum = (num) => {
      console.log("父节点接收到的值: ", num); // 100
    };
    return {
      ...toRefs(state),
      getDemoNum,
    };
  },
};
</script>

    子组件

    <!-- 子节点全部通过props属性获取值 -->
<script>
import { reactive, toRefs } from "vue";
export default {
  name: "Demo",
  props: ["getDemoNum","total"],
  setup(props, context) {
    const state = reactive({
      num: 100,
    });
    const putNum = () => {
      console.log(props.total); // 20
      props.getDemoNum(state.num);
    };
    return {
      ...toRefs(state),
      putNum,
    };
  },
};
</script>

    # 父传子

    # 方式一：自定义事件

    • 父组件通过@xxx自定义事件声明一个方法
    • 子组件通过emits:['xxx']，并通过context.emit('xxx')来出发父组件的事件回调

    父组件

    <template>
  <demo @getDemoNum="getDemoNum" />
</template>

<script>
import { reactive, toRefs } from "vue";
import Demo from "./components/Demo.vue";
export default {
  name: "App",
  components: {
    Demo,
  },
  setup() {
    const state = reactive({});
    const getDemoNum = (num) => {
      console.log("父节点接收到的值: ", num); // 100
    };
    return {
      ...toRefs(state),
      getDemoNum,
    };
  },
};
</script>

    子组件

    <template>
  <button @click="putNum">传值</button>
</template>

<script>
import { reactive, toRefs } from "vue";
export default {
  name: "Demo",
  emits: ["getDemoNum"],
  setup(props, context) {
    const state = reactive({
      num: 100,
    });
    const putNum = () => {
      context.emit("getDemoNum", state.num);
    };
    return {
      ...toRefs(state),
      putNum,
    };
  },
};
</script>

    # 方式二：通过ref属性获取

    父组件

    <template>
  <demo ref="demoRefs" />
</template>

<script>
import { reactive, toRefs, ref } from "vue";
import Demo from "./components/Demo.vue";
export default {
  name: "App",
  components: {
    Demo,
  },
  setup() {
    const demoRefs = ref(null);
		onMounted(()=>{
      console.log(demoRefs.value.num); // 100
    })
    return {
      demoRefs,
    };
  },
};
</script>

    子组件

    <script>
import { reactive, toRefs } from "vue";
export default {
  name: "Demo",
  setup() {
    const state = reactive({
      num: 100,
    });
    return {
      ...toRefs(state),
    };
  },
};
</script>

    # 兄弟组件通信

    # 全局事件总线（mitt）

    • 子组件1通过proxy.$mitt.emit("xxx", index)声明函数

    • 子组件2通过proxy.$mitt.on('xxx',(index)=>{})来触发函数并接受值

    • 声明全局属性

    import mitt from 'mitt'
app.config.globalProperties.$mitt = new mitt()
    • 子组件1
    import { getCurrentInstance } from "vue";
export default {
  setup() {
    let { proxy } = getCurrentInstance();
    const getMessage = (index)=>{
      proxy.$mitt.emit("getIndex", index);
    }
    return {
      getMessage
    };
  },
};
    • 子组件2
    import { getCurrentInstance } from "vue";
export default {
  setup() {
    let { proxy } = getCurrentInstance();
    onMounted(() => {
      proxy.$mitt.on("getIndex", (index) => {
        state.currentIndex = index;
      });
    });
    return {};
  },
};
